Error Types

type t =
  1. | No_native_function of {
    1. kernel_name : string;
    2. context : string;
    }
    (*

    Kernel has no native function but native execution was requested

    *)
  2. | No_ir of {
    1. kernel_name : string;
    }
    (*

    Native-only kernel has no IR (should use lazy exception)

    *)
  3. | Unsupported_arg_type of {
    1. arg_type : string;
    2. reason : string;
    3. context : string;
    }
    (*

    Argument type not supported in native execution

    *)
  4. | Type_conversion_failed of {
    1. from_type : string;
    2. to_type : string;
    3. index : int option;
    4. context : string;
    }
    (*

    Failed to convert between types during argument marshalling

    *)
  5. | Backend_not_found of {
    1. backend : string;
    }
    (*

    Unknown backend name

    *)
  6. | No_source_generation of {
    1. backend : string;
    }
    (*

    Backend does not support source code generation

    *)
  7. | Wrong_backend of {
    1. expected : string;
    2. got : string;
    3. operation : string;
    }
    (*

    Operation only works with specific backend

    *)
