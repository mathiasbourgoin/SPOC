Module Vector_types.Custom_helpers

Helper functions for custom type implementations. These wrap Ctypes operations to provide simpler APIs for PPX-generated code.

val read_float32 : unit Ctypes.ptr -> int -> float

Read a float32 value at byte offset from a void pointer

val write_float32 : unit Ctypes.ptr -> int -> float -> unit

Write a float32 value at byte offset to a void pointer

val read_int32 : unit Ctypes.ptr -> int -> int32

val write_int32 : unit Ctypes.ptr -> int -> int32 -> unit

Write an int32 value at byte offset to a void pointer

val read_int64 : unit Ctypes.ptr -> int -> int64

Write an int64 value at byte offset to a void pointer

val read_float64 : unit Ctypes.ptr -> int -> float
